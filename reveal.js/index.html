JavaScript

TypeScript

            
              const obj = { width: 10, height: 15 };
              const area = obj.width * obj.heigth;
              /* Error: Property 'heigth' does not exist on type
              '{ width: number; height: number; }'. Did you mean 'height'? */

JavaScript

            
              const sum = (a, b) => a + b;

              sum(1, 2); // 3
              sum('1', '2'); // '12'
              sum([], []); // ''

TypeScript

            
              const sum = (a: number, b: number) => a + b;

              sum(1, 2); // 3
              sum('1', '2');
              /* Argument of type 'string' is not assignable to 
              parameter of type 'number'. */
              sum([], []);
              /* Argument of type 'never[]' is not assignable to
              parameter of type 'number'. */

Type & Interface

            
              type Count = number;
              type Stock = 'Yes' | 'No';

              const count : Count = 12;
              const stock : Stock = 'Yes';

              interface Product {
                id : number;
                brand: string;
                price: number;
                count: number;
                stock: Stock;
              };

              const NikeSneakers: Product {
                id: 1,
                brand: 'Nike',
                price: 8990,
                count: 123,
                stock: 'Yes',
              };

Union & Intersection

              const carBrand : CarBrands = 'Audi';
              
              type Point2D = {
                x: number;
                y: number;
              };
              type Point3D = Point2D & {
                z: number;

              let point2d : Point2D = { x: 0, y: 0};
              let point3d : Point3D = { x: 0, y: 0, z: 0};

                name: string;
              };

              };

                age: 20,
              };
              
              type User2 {

              };
